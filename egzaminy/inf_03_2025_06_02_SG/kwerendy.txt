kw2

SELECT nazwa, priorytet
FROM notatki
WHERE nazwa LIKE '%na%';

kw3

SELECT osoby.imie, notatki.nazwa
FROM osoby
JOIN notatki ON osoby.id = notatki.Osoby_id
WHERE notatki.priorytet = 5;

kw4

SELECT osoby.imie, COUNT(notatki.id) AS liczba_notatek FROM osoby LEFT JOIN notatki ON osoby.id = notatki.Osoby_id GROUP BY osoby.imie;
